## 5. Lógica MQL5 que FALTA Implementar

### 5.1 Grid Management (🔴 P0)

El EA debe:
1. Detectar señales con `restrictionType` (ej: "R:10")
2. Calcular niveles de grid según `gridStepPips` y `maxLevels`
3. Abrir órdenes en cada nivel cuando el precio llegue
4. Trackear qué niveles están abiertos

**Algoritmo propuesto:**

```mql5
// Estructura para tracking de grid
struct GridLevel {
   int      level;
   double   price;
   double   lotSize;
   ulong    ticket;
   bool     isOpen;
   datetime openedAt;
};

GridLevel g_gridLevels[10]; // Max 10 niveles
string g_currentSignalId = "";
int g_currentDirection = 0; // 1=BUY, -1=SELL

void ProcessSingleSignal(string signalObj)
{
   // Parsear signal
   string signalId = ParseStringField(signalObj, "id");
   string type = ParseStringField(signalObj, "type");
   string side = ParseStringField(signalObj, "side");
   double price = ParseDoubleField(signalObj, "price");
   int maxLevels = ParseIntField(signalObj, "max_levels");

   if(type == "ENTRY")
   {
      // Inicializar grid
      InitializeGrid(signalId, side, price, maxLevels);
      // Abrir nivel 0
      OpenGridLevel(0, price);
   }
   else if(type == "CLOSE")
   {
      CloseAllGridLevels();
      AckSignal(signalId);
   }
}

void InitializeGrid(string id, string side, double basePrice, int maxLevels)
{
   g_currentSignalId = id;
   g_currentDirection = (side == "BUY") ? 1 : -1;

   // Calcular precios de grid
   double step = g_gridStepPips * 0.01; // Pips a precio

   for(int i = 0; i < maxLevels; i++)
   {
      g_gridLevels[i].level = i;
      g_gridLevels[i].price = basePrice + (i * step * g_currentDirection);
      g_gridLevels[i].lotSize = g_gridLot;
      g_gridLevels[i].isOpen = false;
   }
}

void OnTick()
{
   // Check si toca abrir nuevo nivel
   CheckGridLevels();
}

void CheckGridLevels()
{
   if(g_currentSignalId == "") return;

   double bid = SymbolInfoDouble(EASymbol, SYMBOL_BID);
   double ask = SymbolInfoDouble(EASymbol, SYMBOL_ASK);

   for(int i = 0; i < ArraySize(g_gridLevels); i++)
   {
      if(g_gridLevels[i].isOpen) continue;

      bool shouldOpen = false;
      if(g_currentDirection == 1 && bid >= g_gridLevels[i].price)
         shouldOpen = true;
      else if(g_currentDirection == -1 && ask <= g_gridLevels[i].price)
         shouldOpen = true;

      if(shouldOpen)
         OpenGridLevel(i, g_gridLevels[i].price);
   }
}
```

### 5.2 Trailing Stop Virtual (🔴 P0)

El EA debe:
1. Trackear precio actual vs entrada de cada posición
2. Mover SL virtual cuando el precio avanza `trailingActivate` pips
3. Enviar UPDATE al servidor con nuevo virtualSL

**Algoritmo propuesto:**

```mql5
struct VirtualSL {
   ulong    ticket;
   double   entryPrice;
   double   virtualSL;
   double   highestProfit; // Para trailing
   bool     trailingActivated;
};

VirtualSL g_virtualSLs[100];
int g_virtualSLCount = 0;

void UpdateVirtualStops()
{
   for(int i = 0; i < PositionsTotal(); i++)
   {
      if(PositionSelectByTicket(PositionGetTicket(i)))
      {
         if(PositionGetInteger(POSITION_MAGIC) != MagicNumber)
            continue;

         ulong ticket = PositionGetInteger(POSITION_TICKET);
         double openPrice = PositionGetDouble(POSITION_PRICE_OPEN);
         double currentPrice = PositionGetDouble(POSITION_PRICE_CURRENT);
         long type = PositionGetInteger(POSITION_TYPE);

         // Calcular pips flotantes
         double profitPips = 0.0;
         if(type == POSITION_TYPE_BUY)
            profitPips = (currentPrice - openPrice) / 0.01;
         else
            profitPips = (openPrice - currentPrice) / 0.01;

         // Buscar o crear virtual SL
         int idx = FindVirtualSL(ticket);
         if(idx < 0)
         {
            idx = g_virtualSLCount++;
            g_virtualSLs[idx].ticket = ticket;
            g_virtualSLs[idx].entryPrice = openPrice;
            g_virtualSLs[idx].virtualSL = openPrice;
            g_virtualSLs[idx].highestProfit = 0;
            g_virtualSLs[idx].trailingActivated = false;
         }

         // Activar trailing
         if(profitPips >= g_trailingActivate)
         {
            g_virtualSLs[idx].trailingActivated = true;

            // Calcular nuevo SL
            double newSL;
            if(type == POSITION_TYPE_BUY)
               newSL = currentPrice - (g_trailingBack * 0.01);
            else
               newSL = currentPrice + (g_trailingBack * 0.01);

            // Solo mover si es favorable
            if(type == POSITION_TYPE_BUY && newSL > g_virtualSLs[idx].virtualSL)
               g_virtualSLs[idx].virtualSL = newSL;
            else if(type == POSITION_TYPE_SELL && newSL < g_virtualSLs[idx].virtualSL)
               g_virtualSLs[idx].virtualSL = newSL;

            // Verificar si toca cerrar por virtual SL
            bool shouldClose = false;
            if(type == POSITION_TYPE_BUY && currentPrice <= g_virtualSLs[idx].virtualSL)
               shouldClose = true;
            else if(type == POSITION_TYPE_SELL && currentPrice >= g_virtualSLs[idx].virtualSL)
               shouldClose = true;

            if(shouldClose)
            {
               g_trade.PositionClose(ticket);
               SendTradeEvent("...", "CLOSE", ..., ticket);
            }

            // Enviar UPDATE al servidor
            SendPositionUpdate(ticket, currentPrice, g_virtualSLs[idx].virtualSL);
         }
      }
   }
}
```

## 8. Recomendaciones

1. **Priorizar Grid sobre Trailing:** Sin grid, el EA no puede hacer promedios, que es core del negocio.

2. **Añadir campos a Signal:** Considerar añadir `takeProfitPips` y `stopLossPips` al modelo Signal para que el EA sepa dónde cerrar.

3. **Mover risk management al EA:** El servidor ya tiene daily loss limit, pero el EA debe respetar comandos PAUSE del heartbeat.

4. **Logging mejorado:** Añadir más logs para debugging de grid y trailing.

5. **Modo simulación:** Añadir un modo "paper trading" para probar sin dinero real.
